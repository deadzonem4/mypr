function register_my_menu() {
  register_nav_menu('header-menu',__( 'Top Menu Left' ));
  register_nav_menu('enquire-buttons',__( 'enquire buttons' ));
  register_nav_menu('services-menu',__( 'Home services menu' ));
  register_nav_menu('footer-menu-right',__( 'Footer Right' ));
  register_nav_menu('footer-menu-center',__( 'Footer Center' ));
  register_nav_menu('footer-menu-left',__( 'Footer Left' ));

}add_action( 'init', 'register_my_menu' ); добавяш менютата които използваш в хедъра и футъра

Останалите страници от сайта са създадени от администрацията
използва се sidebar.php за сайтбар на някои страници като пак се добавя в администрацията след като е създаден
файлът. Home.php е шаблона за блог страницата където се визуализират създадените постове.
В този сайт няма такава страница.Single.php е шаблон за детайлния преглед на постовете там е зададено да
визуализира заглавието съдържанието и снимката на поста. Index.php ти е шаблона за останалите страници 
които не са блог страница и начална страница.

В assets/dist се намира css и js, а в assets/images снимките
